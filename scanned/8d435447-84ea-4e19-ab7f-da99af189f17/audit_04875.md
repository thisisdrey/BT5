# Incorrect WETH Deposit in wrapAndDepositWETH Function leads to loss of funds 

## Summary
The InfinityPoolsPeriphery contract has a bug in the wrapAndDepositWETH function which is supposed to help users get back their leftover ETH after a transaction. However, due to using address(this).balance after depositing, the function does not correctly credit the user's WETH balance and the WETH remains locked in the contract. A proof of concept test has been provided to demonstrate the issue and a recommendation has been made to fix the bug by storing address(this).balance in a local variable before the deposit call. It is also recommended to test all functions to prevent similar issues in the future.

## Description
The `InfinityPoolsPeriphery` contract implements a `wrapAndDepositWETH` function. The purpose of the function according to the code comments is as follows:  
This function is to help the user get back the leftover ETH at the end of a transaction.

The `wrapAndDepositWETH` function correctly wraps Ether into WETH using `IWETH9.deposit`, but due to using `address(this).balance` after the deposit, the function will account for 0 WETH instead of the deposited amount. This results in the user receiving 0 WETH despite their Ether being wrapped. This WETH remains permanently locked in the contract.

## Proof of Concept
Add the following test to `test/cantina`:

```solidity
// SPDX-License-Identifier: UNLICENSED
pragma solidity ^0.8.20;

import {InfinityPoolsBase} from "test/cantina/InfinityPoolsBase.t.sol";
import {CorrectInfinityPoolsBase} from "test/cantina/CorrectBase.t.sol";
import {Quad, fromUint256, LibOptQuad} from "src/types/ABDKMathQuad/Quad.sol";
import {Test, console, stdError} from "forge-std/Test.sol";
import {InfinityPool} from "src/InfinityPool.sol";
import {Token} from "src/mock/Token.sol";
import {IInfinityPool} from "src/interfaces/IInfinityPool.sol";
import {InfinityPoolState} from "src/InfinityPoolState.sol";
import {Spot} from "src/libraries/external/Spot.sol";
import {IInfinityPoolsPeriphery} from "src/periphery/interfaces/IInfinityPoolsPeriphery.sol";

contract WrapAndDepositBugPOC is CorrectInfinityPoolsBase {
    address alice;

    function setUp() public override {
        super.setUp();
        // Setup alice
        alice = address(444);
        vm.deal(alice, 10e30);
        token1.mint(alice, 10e30);
        vm.startPrank(alice);
        token1.approve(address(infinityPoolsPeriphery), 10e30);
    }

    function test__AliceLosesFunds() external {
        // Alice adds liquidity to the WETH/token1 pool using ETH
        vm.startPrank(alice);
        // Prepare the AddLiquidityParams struct
        IInfinityPoolsPeriphery.AddLiquidityParams memory params = IInfinityPoolsPeriphery.AddLiquidityParams({
            token0: address(token1),
            token1: address(WETH),
            splits: 15,
            startEdge: 0,
            stopEdge: 2048,
            amount0Desired: 10e18,
            amount1Desired: 10e18,
            amount0Min: 0,
            amount1Min: 0
        });
        // Add liquidity to the pool using the periphery contract
        infinityPoolsPeriphery.addLiquidity{value: 10e18}(params);
        // Log the excess ETH balance in the periphery contract that was not converted
        // to WETH and sent to the pool contract when adding liquidity
        uint256 peripheryExcessEthBalance = address(infinityPoolsPeriphery).balance;
        console.log("peripheryExcessEthBalance =%e", peripheryExcessEthBalance);
        // Alice calls `wrapAndDepositWETH` to wrap it into WETH and deposit it on her behalf
        infinityPoolsPeriphery.wrapAndDepositWETH(alice);
        // The function wrongly credits 0 WETH to her
        // Show that alice's weth balance in the contract is 0
        uint256 aliceWethBalanceInPeriphery = infinityPoolsPeriphery.deposits(alice, address(WETH));
        console.log("aliceWethBalanceInPeriphery =", aliceWethBalanceInPeriphery);
        assertEq(aliceWethBalanceInPeriphery, 0);
        // Show that the contract has WETH that is locked and not
        // attributed to anyone
        uint256 peripheryWethBalance = WETH.balanceOf(address(infinityPoolsPeriphery));
        console.log("peripheryWethBalance =%e", peripheryWethBalance);
        assertEq(peripheryWethBalance, 1e19);
        // The impact is alice lost 1e19 ETH
    }
}
```

## Recommendation
Store `address(this).balance` in a local variable before the deposit call, and use that variable when depositing WETH:

```solidity
function wrapAndDepositWETH(address to) external payable {
    uint256 balance = address(this).balance;
    IWETH9(WETH9).deposit{value: balance}();
    _depositERC20(WETH9, to, balance);
}
```

I would also recommend testing every function since this issue would have been easily caught by a unit test.
